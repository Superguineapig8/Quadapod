pwm.setPWM(FLF, 0, degrees[3]); pwm.setPWM(BRT, 0, degrees[4]); pwm.setPWM(BRF, 0, degrees[5]); pwm.setPWM(BLT, 0, degrees[6]); pwm.setPWM(BLF, 0, degrees[7]); delay(delayTime); } void Fsequence1() //Front Right and Back Left Motor move forward { pwm.setPWM(FRT, 0, degrees[8]); pwm.setPWM(BLF, 0, degrees[23]); delay(delayTime); pwm.setPWM(BLT, 0, degrees[6]); delay(delayTime); pwm.setPWM(BRT, 0, degrees[14]); //Pick up the other two legs slightly pwm.setPWM(FLT, 0, degrees[13]); delay(delayTime); } void Fsequence2() //Front Left and Back Right Motor move forward { pwm.setPWM(FLT, 0, degrees[9]); //Move Front Left Leg delay(delayTime); pwm.setPWM(FLF, 0, degrees[19]); delay(delayTime); pwm.setPWM(BLT, 0, degrees[6]); //Put the tibias back down of the other two legs pwm.setPWM(FRT, 0, degrees[0]); delay(delayTime); } void forward() //Move the front right and back left legs forward first, then the front left and back right legs { Fsequence1(); delay(delayTime); pwm.setPWM(BRT, 0, degrees[14]); //Pick up the other two legs slightly pwm.setPWM(FLT, 0, degrees[13]); delay(delayTime); pwm.setPWM(BLF, 0, degrees[7]); //Move the femurs to base position pwm.setPWM(FRF, 0, degrees[1]); pwm.setPWM(BRT, 0, degrees[4]); //Put the tibias back down of the other two legs pwm.setPWM(FLT, 0, degrees[2]); delay(delayTime); } void Bsequence2() //Front Left and Back Right Motor move forward { pwm.setPWM(FLT, 0, degrees[9]); //Move Front Left Leg delay(delayTime); pwm.setPWM(FLF, 0, degrees[18]); delay(delayTime); pwm.setPWM(FLT, 0, degrees[2]); delay(delayTime); pwm.setPWM(BRT, 0, degrees[10]); // Move Back Right Leg delay(delayTime); delay(delayTime); } void backward() //Move the front right and back left legs forward first, then the front left and back right legs { Bsequence1(); Bsequence2(); } void FrightSide() //Helper method to turn all femurs on the right forward { //Move Front Right Leg first pwm.setPWM(FRT, 0, degrees[8]); //Raise Front Right tibia delay(delayTime); pwm.setPWM(FRF, 0, degrees[17]); //Move Front Right Femur forward delay(delayTime); pwm.setPWM(FRT, 0, degrees[0]); //Lower Front Right Tibia delay(delayTime);
